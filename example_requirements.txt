# 示例：被测项目的 requirements.txt
# 展示如何添加 python-coverage-agent

# ============================================
# 方式1: 从本地路径安装（开发测试推荐）
# ============================================
# -e /path/to/orbit/pca

# ============================================
# 方式2: 从 Git 仓库安装
# ============================================
# git+https://github.com/your-org/orbit.git#subdirectory=pca
# 或指定分支
# git+https://github.com/your-org/orbit.git@main#subdirectory=pca

# ============================================
# 方式3: 从 Wheel 文件安装
# ============================================
# python_coverage_agent @ file:///path/to/python_coverage_agent-0.1.0-py3-none-any.whl

# ============================================
# 方式4: 从 PyPI 安装（如果已发布）
# ============================================
# python-coverage-agent>=0.1.0

# ============================================
# 方式5: 从 HTTP 服务器安装 Wheel
# ============================================
# python_coverage_agent @ http://your-server.com/packages/python_coverage_agent-0.1.0-py3-none-any.whl

# 其他项目依赖...
# flask>=2.0.0
# requests>=2.28.0
